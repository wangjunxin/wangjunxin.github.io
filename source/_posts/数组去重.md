###### 简单数组

```javascript
var a=  [1,2,2,1,3,7,4,6]
var set =  new Set(a)  //  {1,2,3,7,4,6}
var res =  Array.from(set)  //  [1,2,3,7,4,6]
```
###### 对象数组:

```javascript
function unique(list) {
    const result = []
    const obj = {}
    list.forEach((item) => {
        if (!obj[item.id]) {
            result.push(item)
            obj[item.id] = true
        }
    })
    return result
}
```
